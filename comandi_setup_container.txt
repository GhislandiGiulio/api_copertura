DA AGGIUNGERE A NGINX

server {
    listen 443 ssl;
    server_name cltrgoup.it;

    ssl_certificate /etc/letsencrypt/live/cltgroup.it/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/cltgroup.it/privkey.pem;

    location / {
        proxy_pass http://127.0.0.1:6443;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
